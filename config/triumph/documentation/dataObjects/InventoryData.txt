These are the functions which can be called on an InventoryData object.



addItem:
    Arguments:
        ItemBlockData
    Usage:
        item data
    Notes:
        Adds a required item.

setOccupiedMin:
    Arguments:
        float
    Usage:
        min
    Notes:
        Sets the min value for occupied inventory slots.

setEmptyMinMax:
    Arguments:
        float, float
    Usage:
        min, max
    Notes:
        Sets the min and max values for empty inventory slots.

setFullMin:
    Arguments:
        float
    Usage:
        min
    Notes:
        Sets the min value for full inventory slots.

setEmptyMin:
    Arguments:
        float
    Usage:
        min
    Notes:
        Sets the min value for empty inventory slots.

setOccupiedMax:
    Arguments:
        float
    Usage:
        max
    Notes:
        Sets the max value for occupied inventory slots.

setFullMax:
    Arguments:
        float
    Usage:
        max
    Notes:
        Sets the max value for full inventory slots.

setEmptyMax:
    Arguments:
        float
    Usage:
        max
    Notes:
        Sets the max value for empty inventory slots.

setFullMinMax:
    Arguments:
        float, float
    Usage:
        min, max
    Notes:
        Sets the min and max values for full inventory slots.

setOccupiedMinMax:
    Arguments:
        float, float
    Usage:
        min, max
    Notes:
        Sets the min and max values for occupied inventory slots.
